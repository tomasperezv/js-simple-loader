 19 /**
 20  * The Ajax engine performs basic ajax operations. 
 21  */
 22 var AjaxEngine = {
 23 
 24 	STATE_LOADED: 4,
 25 	GET: 'get', 
 26 	POST: 'post',
 27 
 28 	appUrl: '',
 29 	xhrs: [],
 30 
 31 	definition: null,
 32 	definitionPath: null,
 33 
 34 	/**
 35 	 * Depending on the state of the xhr object we can know if it finished pending operations and
 36 	 * reuse it in that case.
 37 	 * @return XMLHttpRequest  
 38 	 */
 39 	getXHR: function() {
 40 
 41 		var xhr = null,
 42 		xhrQueue = this.xhrs,
 43 		length = xhrQueue.length;
 44 		
 45 		for(var i = 0; i < length; i++) {
 46 			if (xhrQueue[i].readyState === this.STATE_LOADED) {
 47 				xhr = xhrQueue[i];
 48 			}
 49 		}
 50 
 51 		if (xhr === null) {
 52 			xhr = new XMLHttpRequest();
 53 			this.xhrs.push(xhr);
 54 		}
 55 		
 56 		return xhr;
 57 	},
 58 
 59 	open: function(method, url, params, onSuccess, onError) {
 60 
 61 		var self = this,
 62 			xhr = this.getXHR(),
 63 			strParams = Utils.URLEncode(params),
 64 			xdr = false;
 65 		
 66 		if (this.appUrl !== '') {
 67 			url = this.appUrl + url;
 68 		}
 69 
 70 		try {
 71 			xhr.open(method, url, true);	
 72 		} catch (e) {
 73 			if (window.XDomainRequest) {
 74 				// Fallback to XDomainRequest
 75 				xhr = new XDomainRequest();
 76 				xdr = true;
 77 				xhr.open(method, url, true);	
 78 			}
 79 		}
 80 
 81 		if (xhr) {
 82 
 83 			if (!xdr) {
 84 				xhr.onreadystatechange = function() { 
 85 					self.processResponse.call(this, onSuccess);
 86 				};
 87 				this.setRequestHeaders(xhr, method, strParams.length);
 88 			} else {
 89 				xhr.onload = function() {
 90 					// The fallback to XDomainRequest only support JSON responses.
 91 					onSuccess( JSON.parse(this.responseText) );
 92 				};
 93 			}
 94 
 95 			xhr.send(strParams);
 96 
 97 		}
 98 	},
 99 
100 	setRequestHeaders: function(xhr, method, length) {
101 		if (method === this.POST) {
102 			xhr.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
103 		}
104 	},
105 
106 	processResponse: function(onSuccess) {
107 		if (this.readyState === AjaxEngine.STATE_LOADED) {
108 			switch(this.status) {
109 				case 200:
110 				case 202:
111 					if (typeof onSuccess !== 'undefined') {
112 						var contentType = this.getResponseHeader('content-type')||'';
113 						if (contentType.indexOf('text/html') >= 0) {
114 							var result = this.responseText;
115 						} else {
116 							var result = JSON.parse(this.responseText);
117 						}
118 						onSuccess(result);
119 					}
120 					break;
121 
122 				default:
123 					if (typeof onError !== 'undefined') {
124 						onError(this);
125 					}
126 					break;
127 			}
128 		}
129 	},
130 
131 	/**
132 	 * Simple post request to a given url. Return the results to the onSuccess
133 	 * callback (as an object).
134 	 *
135 	 * @param String url  
136 	 * @param Object params
137 	 * @param Function onSuccess
138 	 * @param Function onError
139 	 */
140 	post: function(url, params, onSuccess, onError) {
141 		this.open(this.POST, url, params, onSuccess, onError);
142 	},
143 
144 	/**
145 	 * Simple get request to a given url. Return the results to the onSuccess
146 	 * callback (as an object).
147 	 *
148 	 * @param String url  
149 	 * @param Object params  
150 	 * @param Function onSuccess
151 	 * @param Function onError
152 	 */
153 	get: function(url, params, onSuccess, onError) {
154 		this.open(this.GET, url, params, onSuccess, onError);
155 	}
156 
157 };

159 /**
160  *  The loader object is in charge of loading the static files defined in the
161  *  loadDefinition.json file. It uses Ajax requests and dynamic DOM creation of
162  *  elements for injecting the loaded files, for avoiding blocking.
163  *
164  *  Stores a reference to the pending files to be loaded and when all of them
165  *  are ready the onSuccess passed to the loadFiles function is executed.
166  */
167 var Loader = {
168 
169 	CSS: 'css',
170 	JS: 'js',
171 	MAIN_PATH: './js/',
172 
173 	loadQueue: [],
174 	onSuccess: null,
175 	definitionPath: null,
176 	
177 	/**
178 	 * Load the files defined in the json and call to the callback function when
179 	 * everything is ready.
180 	 *
181 	 * @param Function onSuccess  
182 	 */
183 	loadFiles: function(onSuccess) {
184 
185 		var self = this;
186 
187 		if (this.definition == null) {
188 			// First load the file with the definition of the dependences
189 			AjaxEngine.get(this.getDefinitionPath(), {}, function(definition) {
190 				self.definition = definition;
191 				self._load(definition, onSuccess);
192 			});
193 		} else {
194 			this._load(this.definition, onSuccess);
195 		}
196 
197 	},
198 
199 	_load: function(files, onSuccess) {
200 		// Update the load queue and prepare the onSuccess callback
201 		this.loadQueue = this.getLoadQueue(files);
202 		this.onSuccess = onSuccess;
203 
204 		var nFiles = files.length;
205 		for (var i = 0; i < nFiles; i++) {
206 			this.loadFile(files[i]);
207 		}
208 	},
209 
210 	setDefinitionPath: function(definitionPath) {
211 		this.definitionPath = definitionPath;
212 	},
213 
214 	setDefinition: function(definition) {
215 		this.definition = definition;
216 	},
217 
218 	getDefinitionPath: function() {
219 		return this.definitionPath !== null ? this.definitionPath : this.MAIN_PATH + './loadDefinition.json'
220 	},
221 
222 	/**
223 	 * Generates the queue in charge of storing the pending files to be loaded,
224 	 * receiving the array defined in the json definition.
225 	 *
226 	 * @param Array files  
227 	 */
228 	getLoadQueue: function(files) {
229 
230 		var result = [];
231 		var counter = 0;
232 		var filename = '';
233 
234 		var nFiles = files.length;
235 		for (var i = 0; i < nFiles; i++) {
236 			filename = files[i];
237 			if (this.getExtension(filename) === this.JS) {
238 				result[counter] = filename; 
239 				counter++;
240 			}
241 		}
242 
243 		return result;
244 	},
245 	
246 	/**
247 	 * When one of the files is ready we can update the status of the queue, and
248 	 * check if is anything else pending, if not we will call to the callback.
249 	 *
250 	 * @param String file  
251 	 * @param Function callback
252 	 */
253 	updateQueueState: function(file, onSuccess) {
254 
255 		var nFiles = this.loadQueue.length;
256 		var position = -1;
257 
258 		for (var i = 0; i < nFiles; i++) {
259 			if (this.loadQueue[i] === file) {
260 				position = i;
261 				break;
262 			}
263 		}
264 
265 		// Remove the file from the queue
266 		if (position !== -1) {
267 
268 			this.loadQueue.splice(position, 1);
269 
270 			if (this.loadQueue.length === 0) {
271 				onSuccess();
272 			}
273 		}
274 	},
275 
276 	/**
277 	 * Append a dynamic DOM element to the header.
278 	 *
279 	 * @param HTMLDOMElement element  
280 	 */
281 	appendToHead: function(element) {
282 		if (document.getElementsByTagName('head').length > 0) {
283 			document.getElementsByTagName('head')[0].appendChild(element);
284 		}
285 	},
286 
287 	/**
288 	 * Bind the callback for a dynamic script tag.
289 	 *
290 	 * @param HTMLDOMElement element  
291 	 * @param Function callback
292 	 */
293 	attachCallback: function(element, callback) {
294 		if (element.readyState) {
295 			element.onreadystatechange = function() {
296 				if (element.readyState === 'loaded' || element.readyState === 'complete') {
297 					element.onreadystatechange = null;
298 					callback();
299 				}
300 			}
301 		} else {
302 			element.onload = callback;
303 		}
304 	},
305 
306 	/**
307 	 * Add dynamically a script element.
308 	 *
309 	 * @param String filename  
310 	 * @param Function callback
311 	 */
312 	loadJS: function(filename, callback) {
313 
314 		var script = document.createElement('script');
315 		script.type = 'text/javascript';
316 
317 		var self = this;
318 		this.attachCallback(script, function() {
319 			self.updateQueueState(filename, self.onSuccess);
320 		});
321 
322 		script.src = filename;
323 		this.appendToHead(script);
324 	},
325 
326 	/**
327 	 * Add a css dynamically.  
328 	 *
329 	 * @param String filename
330 	 * @param Function callback
331 	 */
332 	loadCSS: function(filename, callback) {
333 
334 		var element = document.createElement('link')
335 
336 		element.setAttribute('rel', 'stylesheet');
337 		element.setAttribute('type', 'text/css');
338 		element.setAttribute('href', filename);
339 
340 		this.appendToHead(element);
341 	},
342 
343 	/**
344 	 * @param String filename  
345 	 * @return String
346 	 */
347 	getExtension: function(filename) {
348 		return filename.split('.').pop().toLowerCase();
349 	},
350 
351 	/**
352 	 * Create a dynamic element for injecting a css or js file in the DOM.  
353 	 */
354 	loadFile: function(filename) {
355 
356 		var extension = this.getExtension(filename);
357 
358 		switch (extension) {
359 
360 			case this.CSS:
361 				this.loadCSS(filename);
362 				break;
363 
364 			case this.JS:
365 				this.loadJS(filename);
366 				break;
367 		}
368 	}
369 };

371 var Utils = {
372 	/**
373 	 * URL-encode parameters passed as an object,
374 	 * e.g. params = {a: 'value'} => 'a=value'
375 	 * @param Object params
376 	 * @author tom@0x101.com  
377 	 */
378 	URLEncode: function(params, separator) {
379 
380 		if (typeof separator === 'undefined') {
381 			var separator = '&';
382 		}
383 
384 		var result = '';
385 		for (var key in params) {
386 			if (params.hasOwnProperty(key)) {
387 				result += key + '=' + params[key] + separator;
388 			}
389 		}
390 
391 		// Remove the last '&' from the string
392 		return result.substr(0, result.length-1);
393 	}
394 };

398  * Basic JSON support for cross-browser compatibility
399  */
400 var JSON = JSON || {}; 
401 JSON.parse = JSON.parse || function (strObject) {  
402 
403 	if (strObject === "") {
404 		var strObject = '""';
405 	}
406 
407 	eval("var result = " + strObject + ";");
408 
409 	return result;
