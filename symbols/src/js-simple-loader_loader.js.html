 19 /**
 20  * The Ajax engine performs basic ajax operations.
 21  * @class AjaxEngine
 22  * @author tom@0x101.com
 23  */
 24 var AjaxEngine = {
 25 
 26 	STATE_LOADED: 4,
 27 	GET: 'get', 
 28 	POST: 'post',
 29 
 30 	appUrl: '',
 31 	xhrs: [],
 32 
 33 	definition: null,
 34 	definitionPath: null,
 35 
 36 	/**
 37 	 * Depending on the state of the xhr object we can know if it finished pending operations and
 38 	 * reuse it in that case.
 39 	 * @return {XMLHttpRequest}
 40 	 */
 41 	getXHR: function() {
 42 
 43 		var xhr = null,
 44 		xhrQueue = this.xhrs,
 45 		length = xhrQueue.length;
 46 		
 47 		for(var i = 0; i < length; i++) {
 48 			if (xhrQueue[i].readyState === this.STATE_LOADED) {
 49 				xhr = xhrQueue[i];
 50 			}
 51 		}
 52 
 53 		if (xhr === null) {
 54 			xhr = new XMLHttpRequest();
 55 			this.xhrs.push(xhr);
 56 		}
 57 		
 58 		return xhr;
 59 	},
 60 
 61 	/**
 62 	 * Opens a XHR POST/GET request
 63 	 * @param {String} method Type of request: POST/GET
 64 	 * @param {String} url
 65 	 * @param {Array} params
 66 	 * @param {Function} onSuccess
 67 	 * @param {Function} onError
 68 	 * @author tom@0x101.com
 69 	 */
 70 	open: function(method, url, params, onSuccess, onError) {
 71 
 72 		var self = this,
 73 			xhr = this.getXHR(),
 74 			strParams = Utils.URLEncode(params),
 75 			xdr = false;
 76 		
 77 		if (this.appUrl !== '') {
 78 			url = this.appUrl + url;
 79 		}
 80 
 81 		try {
 82 			xhr.open(method, url, true);	
 83 		} catch (e) {
 84 			if (window.XDomainRequest) {
 85 				// Fallback to XDomainRequest
 86 				xhr = new XDomainRequest();
 87 				xdr = true;
 88 				xhr.open(method, url, true);	
 89 			}
 90 		}
 91 
 92 		if (xhr) {
 93 
 94 			if (!xdr) {
 95 				xhr.onreadystatechange = function() { 
 96 					self.processResponse.call(this, onSuccess);
 97 				};
 98 				this.setRequestHeaders(xhr, method, strParams.length);
 99 			} else {
100 				xhr.onload = function() {
101 					// The fallback to XDomainRequest only support JSON responses.
102 					onSuccess( JSON.parse(this.responseText) );
103 				};
104 			}
105 
106 			xhr.send(strParams);
107 
108 		}
109 	},
110 
111 	/**
112 	 * @param {XMLHttpRequest} xhr
113 	 * @param {String} method
114 	 * @param {Integer} length
115 	 * @author tom@0x101.com
116 	 */
117 	setRequestHeaders: function(xhr, method, length) {
118 		if (method === this.POST) {
119 			xhr.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
120 		}
121 	},
122 
123 	/**
124 	 * @param {Function} onSuccess
125 	 * @author tom@0x101.com
126 	 */
127 	processResponse: function(onSuccess) {
128 		if (this.readyState === AjaxEngine.STATE_LOADED) {
129 			switch(this.status) {
130 				case 200:
131 				case 202:
132 					if (typeof onSuccess !== 'undefined') {
133 						var contentType = this.getResponseHeader('content-type')||'';
134 						if (contentType.indexOf('text/html') >= 0) {
135 							var result = this.responseText;
136 						} else {
137 							var result = JSON.parse(this.responseText);
138 						}
139 						onSuccess(result);
140 					}
141 					break;
142 
143 				default:
144 					if (typeof onError !== 'undefined') {
145 						onError(this);
146 					}
147 					break;
148 			}
149 		}
150 	},
151 
152 	/**
153 	 * Simple post request to a given url. Return the results to the onSuccess
154 	 * callback (as an object).
155 	 *
156 	 * @param {String} url
157 	 * @param {Object} params
158 	 * @param {Function} onSuccess
159 	 * @param {Function} onError
160 	 * @author tom@0x101.com
161 	 */
162 	post: function(url, params, onSuccess, onError) {
163 		this.open(this.POST, url, params, onSuccess, onError);
164 	},
165 
166 	/**
167 	 * Simple get request to a given url. Return the results to the onSuccess
168 	 * callback (as an object).
169 	 *
170 	 * @param {String} url
171 	 * @param {Object} params
172 	 * @param {Function} onSuccess
173 	 * @param {Function} onError
174 	 * @author tom@0x101.com
175 	 */
176 	get: function(url, params, onSuccess, onError) {
177 		this.open(this.GET, url, params, onSuccess, onError);
178 	}
179 
180 };

182 /**
183  * The loader object is in charge of loading the static files defined in the
184  * loadDefinition.json file. It uses Ajax requests and dynamic DOM creation of
185  * elements for injecting the loaded files, for avoiding blocking.
186  *
187  * Stores a reference to the pending files to be loaded and when all of them
188  * are ready the onSuccess passed to the loadFiles function is executed.
189  * @class Loader
190  * @author tom@0x101.com
191  */
192 var Loader = {
193 
194 	CSS: 'css',
195 	JS: 'js',
196 	MAIN_PATH: './js/',
197 
198 	loadQueue: [],
199 	onSuccess: null,
200 	definitionPath: null,
201 	
202 	/**
203 	 * Load the files defined in the json and call to the callback function when
204 	 * everything is ready.
205 	 *
206 	 * @param {Function} onSuccess
207 	 * @author tom@0x101.com
208 	 */
209 	loadFiles: function(onSuccess) {
210 
211 		var self = this;
212 
213 		if (this.definition == null) {
214 			// First load the file with the definition of the dependences
215 			AjaxEngine.get(this.getDefinitionPath(), {}, function(definition) {
216 				self.definition = definition;
217 				self._load(definition, onSuccess);
218 			});
219 		} else {
220 			this._load(this.definition, onSuccess);
221 		}
222 
223 	},
224 
225 	_load: function(files, onSuccess) {
226 		// Update the load queue and prepare the onSuccess callback
227 		this.loadQueue = this.getLoadQueue(files);
228 		this.onSuccess = onSuccess;
229 
230 		var nFiles = files.length;
231 		for (var i = 0; i < nFiles; i++) {
232 			this.loadFile(files[i]);
233 		}
234 	},
235 
236 	setDefinitionPath: function(definitionPath) {
237 		this.definitionPath = definitionPath;
238 	},
239 
240 	setDefinition: function(definition) {
241 		this.definition = definition;
242 	},
243 
244 	getDefinitionPath: function() {
245 		return this.definitionPath !== null ? this.definitionPath : this.MAIN_PATH + './loadDefinition.json'
246 	},
247 
248 	/**
249 	 * Generates the queue in charge of storing the pending files to be loaded,
250 	 * receiving the array defined in the json definition.
251 	 *
252 	 * @param {Array} files
253 	 * @author tom@0x101.com
254 	 */
255 	getLoadQueue: function(files) {
256 
257 		var result = [];
258 		var counter = 0;
259 		var filename = '';
260 
261 		var nFiles = files.length;
262 		for (var i = 0; i < nFiles; i++) {
263 			filename = files[i];
264 			if (this.getExtension(filename) === this.JS) {
265 				result[counter] = filename; 
266 				counter++;
267 			}
268 		}
269 
270 		return result;
271 	},
272 	
273 	/**
274 	 * When one of the files is ready we can update the status of the queue, and
275 	 * check if is anything else pending, if not we will call to the callback.
276 	 *
277 	 * @param {String} file  
278 	 * @param {Function} callback
279 	 * @author tom@0x101.com
280 	 */
281 	updateQueueState: function(file, onSuccess) {
282 
283 		var nFiles = this.loadQueue.length;
284 		var position = -1;
285 
286 		for (var i = 0; i < nFiles; i++) {
287 			if (this.loadQueue[i] === file) {
288 				position = i;
289 				break;
290 			}
291 		}
292 
293 		// Remove the file from the queue
294 		if (position !== -1) {
295 
296 			this.loadQueue.splice(position, 1);
297 
298 			if (this.loadQueue.length === 0) {
299 				onSuccess();
300 			}
301 		}
302 	},
303 
304 	/**
305 	 * Append a dynamic DOM element to the header.
306 	 *
307 	 * @param {HTMLDOMElement} element
308 	 * @author tom@0x101.com
309 	 */
310 	appendToHead: function(element) {
311 		if (document.getElementsByTagName('head').length > 0) {
312 			document.getElementsByTagName('head')[0].appendChild(element);
313 		}
314 	},
315 
316 	/**
317 	 * Bind the callback for a dynamic script tag.
318 	 *
319 	 * @param {HTMLDOMElement} element  
320 	 * @param {Function} callback
321 	 * @author tom@0x101.com
322 	 */
323 	attachCallback: function(element, callback) {
324 		if (element.readyState) {
325 			element.onreadystatechange = function() {
326 				if (element.readyState === 'loaded' || element.readyState === 'complete') {
327 					element.onreadystatechange = null;
328 					callback();
329 				}
330 			}
331 		} else {
332 			element.onload = callback;
333 		}
334 	},
335 
336 	/**
337 	 * Add dynamically a script element.
338 	 *
339 	 * @param {String} filename  
340 	 * @param {Function} callback
341 	 * @author tom@0x101.com
342 	 */
343 	loadJS: function(filename, callback) {
344 
345 		var script = document.createElement('script');
346 		script.type = 'text/javascript';
347 
348 		var self = this;
349 		this.attachCallback(script, function() {
350 			self.updateQueueState(filename, self.onSuccess);
351 		});
352 
353 		script.src = filename;
354 		this.appendToHead(script);
355 	},
356 
357 	/**
358 	 * Add a css dynamically.  
359 	 *
360 	 * @param {String} filename
361 	 * @param {Function} callback
362 	 * @author tom@0x101.com
363 	 */
364 	loadCSS: function(filename, callback) {
365 
366 		var element = document.createElement('link')
367 
368 		element.setAttribute('rel', 'stylesheet');
369 		element.setAttribute('type', 'text/css');
370 		element.setAttribute('href', filename);
371 
372 		this.appendToHead(element);
373 	},
374 
375 	/**
376 	 * @param {String} filename
377 	 * @return {String}
378 	 * @author tom@0x101.com
379 	 */
380 	getExtension: function(filename) {
381 		return filename.split('.').pop().toLowerCase();
382 	},
383 
384 	/**
385 	 * Create a dynamic element for injecting a css or js file in the DOM.
386 	 * @param {String} filename
387 	 * @author tom@0x101.com
388 	 */
389 	loadFile: function(filename) {
390 
391 		var extension = this.getExtension(filename);
392 
393 		switch (extension) {
394 
395 			case this.CSS:
396 				this.loadCSS(filename);
397 				break;
398 
399 			case this.JS:
400 				this.loadJS(filename);
401 				break;
402 		}
403 	}
404 };

406 /**
407  * @class Utils
408  * @author tom@0x101.com
409  */
410 var Utils = {
411 	/**
412 	 * URL-encode parameters passed as an object,
413 	 * e.g. params = {a: 'value'} => 'a=value'
414 	 * @param {Object} params
415 	 * @author tom@0x101.com 
416 	 */
417 	URLEncode: function(params, separator) {
418 
419 		if (typeof separator === 'undefined') {
420 			var separator = '&';
421 		}
422 
423 		var result = '';
424 		for (var key in params) {
425 			if (params.hasOwnProperty(key)) {
426 				result += key + '=' + params[key] + separator;
427 			}
428 		}
429 
430 		// Remove the last '&' from the string
431 		return result.substr(0, result.length-1);
432 	}
433 };

435 /**
436  * Basic JSON support for cross-browser compatibility
437  * @class JSON
438  * @author tom@0x101.com
439  */
440 var JSON = JSON || {}; 
441 JSON.parse = JSON.parse || function (strObject) {  
442 
443 	if (strObject === "") {
444 		var strObject = '""';
445 	}
446 
447 	eval("var result = " + strObject + ";");
448 
449 	return result;
